451void kls_log(Koliseo *kls, const char *tag, const char *format, ...)
452{
453 if (kls == NULL) {
454 fprintf(stderr, "[KLS] %s(): Passed kls was NULL.\n", __func__);
455 return;
456 }
457 if (kls->conf.kls_verbose_lvl > 0) {
458 va_list args;
459 FILE *fp = kls->conf.kls_log_fp;
460 va_start(args, format);
461 if (fp == NULL) {
462 fprintf(stderr,
463 "[KLS] %s(): Failed opening file to print logs.\n",
464 __func__);
465 } else {
466 time_t now = time(0);
467 const struct tm *mytime = localtime(&now);
468 char timeheader[500];
469 if (strftime(timeheader, sizeof timeheader, "%X", mytime)) {
470 fprintf(fp, "[%-10.10s] [%s] [", tag, timeheader);
471 vfprintf(fp, format, args);
472 fprintf(fp, "]\n");
473 }
474 }
475 va_end(args);
476 }
477}

497Koliseo *kls_new_alloc_ext(ptrdiff_t size, kls_alloc_func alloc_func, KLS_Hooks ext_handlers, void* user)
498#else
499Koliseo *kls_new_alloc_ext_dbg(ptrdiff_t size, kls_alloc_func alloc_func, KLS_Hooks ext_handlers, void* user, Koliseo_Loc loc)
500#endif // KOLISEO_HAS_LOCATE
501{
502 if (size < (ptrdiff_t)sizeof(Koliseo)) {
503#ifndef KOLISEO_HAS_LOCATE
504 fprintf(stderr,
505 "[ERROR] at %s(): invalid requested kls size (%td). Min accepted is: (%td).\n",
506 __func__, size, (ptrdiff_t)sizeof(Koliseo));
507#else
508 fprintf(stderr,
509 "[ERROR] " KLS_Loc_Fmt "%s(): invalid requested kls size (%td). Min accepted is: (%td).\n",
510 KLS_Loc_Arg(loc), __func__, size, (ptrdiff_t)sizeof(Koliseo));
511#endif // KOLISEO_HAS_LOCATE
512 //TODO Is it better to abort the program?
513 return NULL;
514 }
515 void *p = alloc_func(size);
516 if (p) {
517 //sprintf(msg,"Allocated (%li) for new KLS.",size);
518 //kls_log("KLS",msg);
519 char h_size[200];
520 kls_formatSize(size, h_size, sizeof(h_size));
521 Koliseo *kls = p;
522 kls->data = p;
523 kls->size = size;
524 kls->offset = sizeof(*kls);
525 kls->prev_offset = kls->offset;
526 kls->has_temp = 0;
527 kls->t_kls = NULL;
528#ifdef KOLISEO_HAS_REGION
529 kls->reglist_kls = NULL;
530 kls->regs = NULL;
531 kls->max_regions_kls_alloc_basic = 0;
532#endif
533 kls_set_conf(kls, KLS_DEFAULT_CONF);
534 kls->stats = KLS_STATS_DEFAULT;
535 kls->conf.kls_log_fp = stderr;
536 kls->hooks = ext_handlers;
537 kls->extension_data = user;
538#ifdef KLS_DEBUG_CORE
539 kls_log(kls, "KLS", "API Level { %i } -> Allocated (%s) for new KLS.",
540 int_koliseo_version(), h_size);
541 kls_log(kls, "KLS", "KLS offset: { %p }.", kls);
542 kls_log(kls, "KLS", "Allocation begin offset: { %p }.",
543 kls + kls->offset);
544#endif
545
546 if (kls->hooks.on_new_handler != NULL) {
547 // Call on_new extension
548 kls->hooks.on_new_handler(kls);
549 }
550#ifdef KOLISEO_HAS_REGION
551 if (kls->conf.kls_autoset_regions == 1) {
552#ifdef KLS_DEBUG_CORE
553 kls_log(kls, "KLS", "Init of KLS_Region_List for kls.");
554#endif
555
556 KLS_Region *kls_header = (KLS_Region *) malloc(sizeof(KLS_Region));
557 kls_header->begin_offset = 0;
558 kls_header->end_offset = kls->offset;
559 kls_header->size =
560 kls_header->end_offset - kls_header->begin_offset;
561 kls_header->padding = 0;
562 kls_header->type = KLS_Header;
563 strncpy(kls_header->name, "KLS_Header", KLS_REGION_MAX_NAME_SIZE);
564 kls_header->name[KLS_REGION_MAX_NAME_SIZE] = '\0';
565 strncpy(kls_header->desc, "Sizeof Koliseo header",
566 KLS_REGION_MAX_DESC_SIZE);
567 kls_header->desc[KLS_REGION_MAX_DESC_SIZE] = '\0';
568 //KLS_Region_List reglist = kls_emptyList();
569 //reglist = kls_cons(kls,kls_header,reglist);
570 //kls->regs = reglist;
571 kls->regs = kls_rl_cons(kls, kls_header, kls->regs);
572 if (kls->regs == NULL) {
573 fprintf(stderr,
574 "[KLS] [%s()]: failed to get a KLS_Region_List.\n",
575 __func__);
576 exit(EXIT_FAILURE);
577 }
578 } else {
579 kls->regs = NULL;
580 }
581#endif // KOLISEO_HAS_REGION
582 } else {
583#ifndef KOLISEO_HAS_LOCATE
584 fprintf(stderr, "[KLS] Failed %s() call.\n", __func__);
585#else
586 fprintf(stderr, "[KLS] " KLS_Loc_Fmt "Failed %s() call.\n", KLS_Loc_Arg(loc), __func__);
587#endif // KOLISEO_HAS_LOCATE
588 exit(EXIT_FAILURE);
589 }
590#ifdef KLS_DEBUG_CORE
591 Koliseo *kls_ref = p;
592 if (kls_ref->conf.kls_verbose_lvl > 0) {
593 print_kls_2file(kls_ref->conf.kls_log_fp, p);
594 }
595#endif
596 return p;
597}
